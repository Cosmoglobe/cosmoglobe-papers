   \title{\bfseries{\Cosmoglobe\ DR3. High resolution thermal Sunyaev-Zeldovich maps from joint analysis of Planck and AKARI}}

   \input{authors.tex}

   \titlerunning{\Cosmoglobe: AKARI reanalysis}
   \authorrunning{R.M.~Sullivan et al.}

   \date{\today} 
   
   \abstract{We present the first all-sky $y$-maps of the thermal Sunyaev-Zeldovich (tSZ) effect from the Commander3 pipeline, using a joint analysis of the \Planck\ PR3 HFI maps and the AKARI maps. We also include the DIRBE maps as produces from the Cosmoglobe DR2 analysis, FIRAS, \Planck\ LFI, and other maps to better constrain the foregrounds. This is part of a joint release of the Cosmoglobe DR3 AKARI data analysis. We additionally provide an updated catalogue of tSZ point sources, and the $y$-map angular power spectrum. The joint analysis additionally releases new CIB and dust maps in separate papers. This joint analysis and component separations reduces the foreground contamination in the final $y$-maps and thus we provide the cleanest $y$-maps from the \Planck\ data release to date.}

   \keywords{ISM: general - Zodiacal dust, Interplanetary medium - Cosmology: observations, diffuse radiation - Galaxy: general}
